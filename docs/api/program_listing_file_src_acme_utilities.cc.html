  ostream_type &
  operator<<(
      ostream_type &os,
      line const &obj)
  {
    vec3 obj_origin(obj.origin());
    vec3 obj_direction(obj.direction());
    os << std::scientific
       << std::showpoint
       << std::setprecision(6)
       << std::endl
       << "Origin = "
       << "[ " << obj_origin[0] << ", " << obj_origin[1] << ", " << obj_origin[2] << " ]'"
       << std::endl
       << "Direction = "
       << "[ " << obj_direction[0] << ", " << obj_direction[2] << ", " << obj_direction[2] << " ]'";
    return os;
  }
