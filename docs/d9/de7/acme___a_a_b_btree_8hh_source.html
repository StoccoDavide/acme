107  template <typename COLLISION_FUNCTION>
108  bool
109  collision(
110  AABBtree const &tree,
111  COLLISION_FUNCTION fun,
112  bool swap_tree = false
113  ) const
114  {
115 
116  // check box with
117  if (!tree.ptrbox->collision(*ptrbox))
118  return false;
119 
120  int icase = (children.empty() ? 0 : 1) +
121  (tree.children.empty() ? 0 : 2);
122 
123  switch (icase)
124  {
125  case 0: // both leaf, use box intersection algorithm
126  if (swap_tree)
127  return fun(tree.ptrbox, ptrbox);
128  else
129  return fun(ptrbox, tree.ptrbox);
130  case 1: // first is a tree, second is a leaf
131  {
132  typename std::vector<AABBtree::ptr>::const_iterator it;
133  for (it = children.begin(); it != children.end(); ++it)
134  if (tree.collision(**it, fun, !swap_tree))
135  return true;
136  }
137  break;
138  case 2: // first leaf, second is a tree
139  {
140  typename std::vector<AABBtree::ptr>::const_iterator it;
141  for (it = tree.children.begin();
142  it != tree.children.end(); ++it)
143  if (this->collision(**it, fun, swap_tree))
144  return true;
145  }
146  break;
147  case 3: // first is a tree, second is a tree
148  {
149  typename std::vector<AABBtree::ptr>::const_iterator it1;
150  typename std::vector<AABBtree::ptr>::const_iterator it2;
151  for (it1 = children.begin(); it1 != children.end(); ++it1)
152  for (it2 = tree.children.begin();
153  it2 != tree.children.end(); ++it2)
154  if ((*it1)->collision(**it2, fun, swap_tree))
155  return true;
156  }
157  break;
158  }
159  return false;
160  }
