      Vertex1 = this.getVertex1().get();
      Vertex2 = this.getVertex2().get();
      scatter3(Vertex1(1), Vertex1(2), Vertex1(3), color, 'filled');
      scatter3(Vertex2(1), Vertex2(2), Vertex2(3), color, 'filled');
      X = [Vertex1(1), Vertex2(1)];
      Y = [Vertex1(2), Vertex2(2)];
